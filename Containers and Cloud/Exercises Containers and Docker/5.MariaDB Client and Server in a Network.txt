1. creating a network:
docker network create some-network
2. creating mariadb_server container
docker run -d --network some-network \
 --name some-mariadb \
 --env MARIADB_USER=example-user \
 --env MARIADB_PASSWORD=my_cool_secret \
 --env MARIADB_ROOT_PASSWORD=my-secret-pw \
 mariadb:latest
3. creating mariadb_client container
docker run -it \
--network some-network \
--rm mariadb mariadb \
-hsome-mariadb \
-uexample-user -p
